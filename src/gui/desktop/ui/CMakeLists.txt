
include_directories($<TARGET_PROPERTY:core,INTERFACE_INCLUDE_DIRECTORIES>

)

    configuration_dialog.cpp

    mainwindow.hpp
)

    configuration_dialog.ui

)

set_target_properties(gui_ui PROPERTIES POSITION_INDEPENDENT_CODE ON)
